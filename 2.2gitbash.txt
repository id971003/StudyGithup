#리눅스 #유닉스 #WINDOW에서리눅스커맨드이용가능

환경설정
git config user.name "name"
git config user.email "email"


로컬리포지토리생성
git clone "HttpPath"

add
git add "FilePath"

커밋
git committ -m "커밋메시지"

푸쉬
git push

풀
git pull

상태보기
git status

add 시킨 거 다시 빼기
git reset

다시 로컬리포지토리로 돌리기
git checkout -- "filename"

커밋메시지 수정
git commit --amend
